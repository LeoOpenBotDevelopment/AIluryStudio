    <script>
        const btn = document.getElementById('theme-toggle');
        const html = document.documentElement;

        // On Load
        const saved = localStorage.getItem('dream-theme');
        if (saved) html.setAttribute('data-theme', saved);

        btn.addEventListener('click', () => {
            const isDark = html.getAttribute('data-theme') === 'dark';
            const next = isDark ? 'light' : 'dark';
            
            html.setAttribute('data-theme', next);
            localStorage.setItem('dream-theme', next);
            
            // Haptic-style animation for the button
            btn.style.transform = 'scale(0.8)';
            setTimeout(() => btn.style.transform = 'scale(1)', 100);
        });

        // Subtle parallax effect for bento cards
        document.addEventListener('mousemove', (e) => {
            const amount = 5;
            const x = (e.clientX / window.innerWidth - 0.5) * amount;
            const y = (e.clientY / window.innerHeight - 0.5) * amount;
            document.querySelector('.hero h1').style.transform = `translate(${x}px, ${y}px)`;
        });
    </script>
