    <script>
        // --- Custom Cursor Logic ---
        const cursor = document.getElementById('cursor');
        document.addEventListener('mousemove', (e) => {
            cursor.style.left = e.clientX + 'px';
            cursor.style.top = e.clientY + 'px';
        });

        // --- Three.js Patterned Ribbon ---
        const scene = new THREE.Scene();
        const camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000);
        const renderer = new THREE.WebGLRenderer({ canvas: document.getElementById('canvas-bg'), alpha: true, antialias: true });
        renderer.setSize(window.innerWidth, window.innerHeight);

        // Unique Twist: A mathematical "Mobius" Ribbon
        const curve = new THREE.TorusKnotGeometry(10, 3, 200, 32);
        const material = new THREE.MeshPhongMaterial({
            color: 0xffffff,
            wireframe: true,
            emissive: 0x222222,
            shininess: 100
        });

        const mesh = new THREE.Mesh(curve, material);
        scene.add(mesh);

        // Lighting to show off patterns
        const pLight = new THREE.PointLight(0xff3e00, 2, 50);
        pLight.position.set(5, 5, 5);
        scene.add(pLight);

        const sLight = new THREE.PointLight(0x00ffff, 2, 50);
        sLight.position.set(-5, -5, 5);
        scene.add(sLight);

        camera.position.z = 25;

        // Scroll Logic: Change Pattern Rhythm
        let currentScroll = 0;
        window.addEventListener('scroll', () => {
            currentScroll = window.scrollY;
            
            // Twist the geometry based on scroll
            mesh.rotation.x = currentScroll * 0.002;
            mesh.rotation.y = currentScroll * 0.001;

            // Change background colors based on section
            const section = Math.floor(currentScroll / window.innerHeight);
            const colors = ['#0a0a0a', '#1a0a2e', '#2e1a0a'];
            document.body.style.backgroundColor = colors[section % colors.length];
        });

        function animate() {
            requestAnimationFrame(animate);
            
            const time = Date.now() * 0.001;
            
            mesh.rotation.z += 0.005;
            
            // Make lights pulse
            pLight.intensity = 1 + Math.sin(time) * 0.5;
            sLight.intensity = 1 + Math.cos(time) * 0.5;

            renderer.render(scene, camera);
        }

        window.addEventListener('resize', () => {
            camera.aspect = window.innerWidth / window.innerHeight;
            camera.updateProjectionMatrix();
            renderer.setSize(window.innerWidth, window.innerHeight);
        });

        animate();
    </script>
