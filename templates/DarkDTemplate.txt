    <script>
        // --- Three.js Monumental Setup ---
        const scene = new THREE.Scene();
        const camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000);
        const renderer = new THREE.WebGLRenderer({ canvas: document.getElementById('canvas-bg'), antialias: true, alpha: true });
        renderer.setSize(window.innerWidth, window.innerHeight);
        renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));

        // Group for monumental objects
        const objectGroup = new THREE.Group();
        scene.add(objectGroup);

        // Lights (Essential for the "Liquid Chrome" look)
        const light1 = new THREE.DirectionalLight(0xffffff, 1);
        light1.position.set(5, 5, 5);
        scene.add(light1);

        const light2 = new THREE.PointLight(0xff4d00, 2); // Orange Accent Light
        light2.position.set(-10, -5, 0);
        scene.add(light2);

        const ambient = new THREE.AmbientLight(0xffffff, 0.2);
        scene.add(ambient);

        // Huge Sculptures
        const geometries = [
            new THREE.TorusKnotGeometry(4, 1.2, 128, 32),
            new THREE.IcosahedronGeometry(5, 0),
            new THREE.TorusGeometry(6, 2, 16, 100),
            new THREE.OctahedronGeometry(4, 0)
        ];

        const objects = [];
        for(let i = 0; i < 6; i++) {
            // Shiny Metallic Material
            const mat = new THREE.MeshStandardMaterial({ 
                color: 0x333333, 
                metalness: 1, 
                roughness: 0.1, 
                wireframe: i % 2 === 0 // Mix wireframes and solid chrome
            });
            
            const mesh = new THREE.Mesh(geometries[i % geometries.length], mat);
            
            // Strategic placement to interact with text sections
            mesh.position.set(
                (i % 2 === 0 ? -1 : 1) * (Math.random() * 5 + 8), // Alternate left/right
                -i * 25, // Spread down the vertical axis
                -5
            );
            
            mesh.rotation.set(Math.random() * 5, Math.random() * 5, Math.random() * 5);
            objectGroup.add(mesh);
            objects.push(mesh);
        }

        camera.position.z = 20;

        // --- Smooth Parallax Scroll ---
        let currentScroll = 0;
        window.addEventListener('scroll', () => {
            currentScroll = window.scrollY;
        });

        function animate() {
            requestAnimationFrame(animate);
            
            // Smoothly move the whole group based on scroll
            const targetY = (currentScroll * 0.05);
            objectGroup.position.y = THREE.MathUtils.lerp(objectGroup.position.y, targetY, 0.1);

            // Independent rotations for each piece
            objects.forEach((obj, idx) => {
                obj.rotation.x += 0.005;
                obj.rotation.y += 0.008;
                // Add a gentle "floating" bounce
                obj.position.y += Math.sin(Date.now() * 0.001 + idx) * 0.01;
            });

            renderer.render(scene, camera);
        }

        window.addEventListener('resize', () => {
            camera.aspect = window.innerWidth / window.innerHeight;
            camera.updateProjectionMatrix();
            renderer.setSize(window.innerWidth, window.innerHeight);
        });

        animate();
    </script>
